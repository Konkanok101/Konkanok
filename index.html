คำสั่งวนซ้ำ: while loop

หน้าแรก ความหมายของ While Loop การทำงานของลูป While บทเรียนที่ 2 ติดต่อเรา

ความหมาย: คำสั่งวนซ้ำ while loop

คำสั่ง while ในภาษา C ใช้สำหรับวนซ้ำการทำงานของชุดคำสั่ง เมื่อเงื่อนไขที่กำหนดยังคงเป็นจริง (true) โดยรูปแบบของคำสั่งมีดังนี้:

while (เงื่อนไข) { // ชุดคำสั่งที่ต้องการให้วนซ้ำ }

ตัวอย่างการใช้งาน while loop

#include <stdio.h> int main() {
    int i = 1; // กำหนดตัวแปรเริ่มต้น
    while (i <= 5) { // เงื่อนไข: วนซ้ำจนกว่า i จะมากกว่า 5
    printf("รอบที่ %d\n", i); // แสดงค่าปัจจุบันของ i
    i++; // เพิ่มค่าของ i ครั้งละ 1
}
    return 0;
  }

ตัวอย่างโปรแกรม: รับค่าจากผู้ใช้จนกว่าจะป้อนเลข 0

#include <stdio.h>
int main() {
    int num;
    printf("ป้อนตัวเลข (0 เพื่อออกจากโปรแกรม): ");
    scanf("%d", &num);

    while (num != 0) { // วนซ้ำจนกว่าผู้ใช้จะป้อน 0
    printf("คุณป้อนเลข: %d\n", num);
    printf("ป้อนตัวเลขอีกครั้ง (0 เพื่อออก): ");
    scanf("%d", &num);
}
    printf("โปรแกรมจบการทำงาน\n");
    return 0;
}
